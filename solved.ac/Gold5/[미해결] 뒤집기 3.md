# 뒤집기 3

[백준 문제 링크](https://www.acmicpc.net/problem/1464)

체점중 21%에서 오답이 발생한다. 접근 방법이 잟못 된건지, 예외 케이스를 못 잡은건지 아직 모르겠다.

```javascript
const fs = require("fs");
const input = fs.readFileSync("/dev/stdin", "utf8").trim();

/**
 * @description 뒤집기 방법으로 만들 수 있는 문자열 중 사전순으로 제일 뒤에있는 것을 출력하는 함수
 * @param {string} S 문자열
 * @param {number} min 제일 앞서는 문자 아스키 코드
 * @param {number} max 제일 뒤에있는 문자 아스키 코드
 * @returns {string}
 */
function getLastString(S, min = 65, max = 90) {
  if (S.length < 2 || max < min) return S;

  const char = String.fromCharCode(max);
  const index = S.indexOf(char);
  if (index === -1) return getLastString(S, min, max - 1);

  let left = "";
  let right = "";

  for (let i = index; i < S.length; i += 1) {
    if (S[i] === char) left += S[i];
    else right += S[i];
  }

  let mid = getFirstString(S.slice(0, index), min, max - 1);
  mid = mid.split("").reverse().join("");

  return left + mid + right;
}

/**
 * @description 뒤집기 방법으로 만들 수 있는 문자열 중 사전순으로 제일 앞서는 것을 출력하는 함수
 * @param {string} S 문자열
 * @param {number} min 제일 앞서는 문자 아스키 코드
 * @param {number} max 제일 뒤에있는 문자 아스키 코드
 * @returns {string}
 */
function getFirstString(S, min = 65, max = 90) {
  if (S.length < 2 || min > max) return S;

  const char = String.fromCharCode(min);
  const index = S.indexOf(char);
  if (index === -1) return getFirstString(S, min + 1, max);

  let left = "";
  let right = "";

  for (let i = index; i < S.length; i += 1) {
    if (S[i] === char) left += S[i];
    else right += S[i];
  }

  let mid = getLastString(S.slice(0, index), min + 1, max);
  mid = mid.split("").reverse().join("");

  return left + mid + right;
}

console.log(getFirstString(input));
```
