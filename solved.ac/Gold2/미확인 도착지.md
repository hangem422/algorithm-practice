# 미확인 도착지

[백준 문제 링크](https://www.acmicpc.net/problem/9370)

```javascript
const fs = require("fs");
const input = fs.readFileSync("/dev/stdin", "utf8").trim().split("\n");

class Heap {
  constructor(comp) {
    this.items = [];
    this.comp = comp;
  }

  get size() {
    return this.items.length;
  }

  swap(a, b) {
    const temp = this.items[a];
    this.items[a] = this.items[b];
    this.items[b] = temp;
  }

  getParentIndex(index) {
    return Math.floor((index - 1) / 2);
  }

  getLeftChildIndex(index) {
    return index * 2 + 1;
  }

  getRightChildIndex(index) {
    return index * 2 + 2;
  }

  peak() {
    return this.items[0];
  }

  add(item) {
    let index = this.items.push(item) - 1;
    let parentIndex = this.getParentIndex(index);

    while (
      parentIndex >= 0 &&
      this.comp(this.items[index], this.items[parentIndex])
    ) {
      this.swap(index, parentIndex);
      index = parentIndex;
      parentIndex = this.getParentIndex(index);
    }
  }

  poll() {
    if (this.size < 2) return this.items.pop();

    const item = this.peak();
    this.items[0] = this.items.pop();

    let index = 0;
    let leftIndex = this.getLeftChildIndex(index);
    let rightIndex = this.getRightChildIndex(index);

    while (leftIndex < this.size) {
      const target =
        rightIndex < this.size &&
        this.comp(this.items[rightIndex], this.items[leftIndex])
          ? rightIndex
          : leftIndex;

      if (this.comp(this.items[index], this.items[target])) break;
      this.swap(index, target);

      index = target;
      leftIndex = this.getLeftChildIndex(index);
      rightIndex = this.getRightChildIndex(index);
    }

    return item;
  }
}

class Graph {
  constructor() {
    this.edges = new Map();
  }

  addVertex(ver) {
    this.edges.set(ver, new Map());
  }

  addEdge(ve1, ver2, weight) {
    this.edges.get(ve1).set(ver2, weight);
    this.edges.get(ver2).set(ve1, weight);
  }

  removeEdge(ve1, ver2) {
    this.edges.get(ve1).delete(ver2);
    this.edges.get(ver2).delete(ve1);
  }

  dijkstra(source) {
    if (!this.edges.has(source)) return;

    const dist = new Map();
    const heap = new Heap((a, b) => a[1] < b[1]);

    this.edges.forEach((_, ver) => dist.set(ver, Infinity));
    dist.set(source, 0);
    heap.add([source, 0]);

    while (heap.size > 0) {
      const [minVer, minDist] = heap.poll();
      if (dist.get(minVer) < minDist) continue;

      this.edges.get(minVer).forEach((_, neighbor) => {
        const alt = minDist + this.edges.get(minVer).get(neighbor);
        if (alt < dist.get(neighbor)) {
          dist.set(neighbor, alt);
          heap.add([neighbor, alt]);
        }
      });
    }

    return dist;
  }
}

function filterCandidates(n, s, g, h, edges, cans) {
  const graph = new Graph();

  for (let i = 1; i <= n; i += 1) {
    graph.addVertex(i);
  }

  edges.forEach((param) => graph.addEdge(...param));
  const fromS = graph.dijkstra(s);
  const fromG = graph.dijkstra(g);
  const fromH = graph.dijkstra(h);

  const res = cans.filter((ver) => {
    if (!isFinite(fromS.get(ver))) return false;
    return (
      fromS.get(g) + fromG.get(h) + fromH.get(ver) === fromS.get(ver) ||
      fromS.get(h) + fromH.get(g) + fromG.get(ver) === fromS.get(ver)
    );
  });

  return res.sort((a, b) => a - b);
}

function solution() {
  const T = +input[0];
  let ans = "";

  for (let i = 0, j = 1; i < T; i += 1) {
    const [n, m, t] = input[j].split(" ").map(Number);
    const [s, g, h] = input[j + 1].split(" ").map(Number);
    const edges = input
      .slice(j + 2, j + 2 + m)
      .map((str) => str.split(" ").map(Number));
    const cans = input.slice(j + 2 + m, j + 2 + m + t).map(Number);

    const res = filterCandidates(n, s, g, h, edges, cans);
    ans += `${res.join(" ")}\n`;
    j += 2 + m + t;
  }

  return ans;
}

console.log(solution());
```
